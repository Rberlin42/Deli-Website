/announcements
    GET: return all announcements
    POST: Update announcements

    Update schema:
    {
        update: [{id: 9, position: 1, title: 'test', description: ''}]
        add: [{position: 1, title: 'test', description: ''}]
        delete: [1, 2, 4, ...]
    }

    return new ids

/contact
    GET: return all active emails (or archived)
    POST: add new email
    DELETE{id}: archive an email

/other
    GET{id}: get other item
    PUT{id}: update other item

/menuitem
    GET{type}: get menu items of type (nested section json)
    POST: add new menu item
    PUT{id}: update menu item
    DELETE{id}: remove a menu item

/menusection
    POST: add a new section
    PUT{id}: update a section
    DELETE{id}: delete a section and all of its items
